SELECT 
	CASE WHEN A.LOGINSTANC = 0 
	THEN 99999 
	ELSE A.LOGINSTANC END AS LOGINSTANC, 
	A.DOCENTRY, A.LINENUM, A.ITEMCODE, A.QUANTITY
	FROM
	(SELECT LOGINSTANC, DOCENTRY, LINENUM, ITEMCODE, QUANTITY FROM RDR1 WHERE ObjType = 17 AND DocEntry = 1
	UNION ALL
	SELECT  LOGINSTANC, DOCENTRY, LINENUM, ITEMCODE, QUANTITY FROM ADO1 WHERE ObjType = 17 AND DocEntry = 1) AS A
	ORDER BY CASE WHEN A.LogInstanc = 0 THEN 99999 ELSE A.LogInstanc END
